program perform;

function agent_inference(task: Task) {
	var infer = streamwait print(type: "infer", data: task, instruction: #"Perform given task: {task}";
	return infer;
}

Task: structure* {
	Data: string;
	Instruction: string;
}

Tasks: structure*<list<Task>>;

Queue: structure*<queue<limit: 8>>;

procedure Main() {
	var agents = [];
	
	var task: Task = streamwait read();
	
	var tasks: Tasks = await print(type: "infer", data: task, 
instruction: #"Plan and split {task} to subtasks", output: Tasks);

	var queue: Queue;
	
	for streamwait by record (tasks, task) {
		await queue += task;
		for streamwait by record (queue, enqueuedTask) {
			var agentResult = streamwait agent_inference(enqueuedTask);
			streamwait print(agentResult);
		}
	}
}

program agent;
use perform;

procedure Main() {
	var <task: Task, instruction: string> = streamwait read();
	var result = await print(type: "infer", device: "agent", control: "infer_control");
	print(result);
	streamwait print(result);
}

		streamwait print(data: #"Performed task: {data}");
